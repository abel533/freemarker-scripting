16  package org.mybatis.scripting.freemarker;
17  
18  import freemarker.core.Environment;
19  import freemarker.ext.util.WrapperTemplateModel;
20  import freemarker.template.*;
21  
22  import java.io.IOException;
23  import java.util.List;
24  import java.util.Map;
25  
26  /**
27   * Custom FreeMarker directive for generating "#{paramName}" declarations in convenient way.
28   * Problem is FreeMarker supports this syntax natively and there are no chance to disable this
29   * (although it is deprecated). And to get "#{paramName}" we should write ${r"#{paramName}"}.
30   * With this directive you can write more simple:
31   *
32   * <blockquote><pre>
33   *     &lt;@p name="paramName"/&gt;
34   * </pre></blockquote>
35   *
36   * Also directive supports `value` attribute. If it is specified, param will take passed value
37   * and create the corresponding #{}-parameter. This is useful in loops:
38   *
39   * <blockquote><pre>
40   *     &lt;#list ids as id&gt;
41   *       &lt;@p value=id/&gt;
42   *       &lt;#if id_has_next&gt;,&lt;/#if&gt;
43   *     &lt;/#list&gt;
44   * </pre></blockquote>
45   *
46   * will be translated into
47   *
48   * <blockquote><pre>
49   *     #{_p0},#{_p1},#{_p2}
50   * </pre></blockquote>
51   *
52   * And MyBatis engine will convert it to `?`-params finally.
53   *
54   * @author elwood
55   */
56  public class MyBatisParamDirective implements TemplateDirectiveModel {
57      public static String DEFAULT_KEY = "p";
58  
59      @Override
60      public void execute(Environment env, Map params, TemplateModel[] loopVars, TemplateDirectiveBody body) throws TemplateException, IOException {
61          SimpleScalar name = (SimpleScalar) params.get("name");
62          if (params.containsKey("value")) {
63              Object valueObject = params.get("value");
64              Object value;
65              if (valueObject == null) {
66                  value = null;
67              } else if (valueObject instanceof WrapperTemplateModel) {
68                  value = ((WrapperTemplateModel) valueObject).getWrappedObject();
69              } else if (valueObject instanceof TemplateScalarModel) {
70                  value = ((TemplateScalarModel) valueObject).getAsString();
71              } else if (valueObject instanceof TemplateNumberModel) {
72                  value = ((TemplateNumberModel) valueObject).getAsNumber();
73              } else if (valueObject instanceof TemplateDateModel) {
74                  value = ((TemplateDateModel) valueObject).getAsDate();
75              } else if (valueObject instanceof TemplateBooleanModel) {
76                  value = ((TemplateBooleanModel) valueObject).getAsBoolean();
77              } else {
78                  throw new UnsupportedOperationException(
79                          String.format("Type %s is not supported yet in this context.",
80                                  valueObject.getClass().getSimpleName()));
81              }
82  
83              TemplateModel generatedParamsObject = env.getGlobalVariables().get(FreeMarkerSqlSource.GENERATED_PARAMS_KEY);
84              List generatedParams;
85              if (generatedParamsObject instanceof DefaultListAdapter) {
86                  generatedParams = (List) ((DefaultListAdapter) generatedParamsObject).getWrappedObject();
87              } else {
88                  generatedParams = ((GeneratedParamsTemplateModel) generatedParamsObject)
89                          .getGeneratedParams();
90              }
91              String generatedParamName = "_p" + generatedParams.size();
92              env.getOut().write(String.format("#{%s}", generatedParamName));
93              generatedParams.add(value);
94          } else {
95              env.getOut().write(String.format("#{%s}", name));
96          }
97      }
98  }
